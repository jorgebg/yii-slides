Active Record

Yii Framework


<?php
#New row
$model = new Lightsaber;
$model->color = 'blue';
$model->save();

#Update row
$model = Lightsaber::model()->find('color=:color', array(':color'=>'blue'));
$model->color = 'red';
$model->save();

#Delete row
$model = Lightsaber::model()->findByAttributes(array('color'=>'red'));
$model->delete();
